##### 最小栈
* 设计一个支持 push ，pop ，top 操作，并能在常数时间内检索到最小元素的栈。
* 栈和树应该先初始化数据结构，def __init__(self)
* 数据结构那可以放一个最小栈min_stack，pop()的时候，append(min(x,min_stack[-1]))

```python
from typing import List
class MinStack:
    def __init__(self):
        self.items = []   # 数据结构使用一个列表

    def push(self, x: int) -> None:
        self.items.append(x)
        print(self.items)

    def pop(self) -> None:
        if len(self.items) > 0:
            self.items.pop()
            print(self.items)

    def top(self) -> int:
        if len(self.items) > 0:
            print(self.items[-1])

    def getMin(self) -> int:
        if len(self.items) > 0:
            min_item = float('inf')
            for i in self.items:
                if i < min_item:
                    min_item = i
            print(min_item)

minstack =  MinStack()
minstack.push(-2)
minstack.push(0)
minstack.push(-3)

minstack.getMin()
minstack.pop()
minstack.top()
minstack.getMin()
```

```python
from typing import List
class MyQueue:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.queue = []


    def push(self, x: int) -> None:
        """
        Push element x to the back of queue.
        """
        self.queue.append(x)


    def pop(self) -> int:
        """
        Removes the element from in front of queue and returns that element.
        """
        return self.queue.pop(0)


    def peek(self) -> int:
        """
        Get the front element.
        """
        return self.queue[0]



    def empty(self) -> bool:
        """
        Returns whether the queue is empty.
        """
        return not bool(self.queue)

queue = MyQueue()

print(queue.push(1))
print(queue.push(2))
print(queue.peek())
print(queue.pop() )
print(queue.empty())
```

##### 用两个栈实现队列
* 使用的是两个栈的原理
```python
from typing import List
class CQueue:

    def __init__(self):
        self.stack1 = []
        self.stack2 = []

    def appendTail(self, value: int) -> None:
        self.stack1.append(value)


    def deleteHead(self) -> int:
        if self.stack2: return self.stack2.pop()
        if not self.stack1: return -1
        while self.stack1:
            self.stack2.append(self.stack1.pop())
        return self.stack2.pop()
```

```python
from typing import List

class ListNode:
    def __init__(self, x):
        self.val = x # 生成的新节点作为头节点
        self.next = None # 头结点的指针域置空

class linklist:
    def __init__(self):  #根据节点的数据结构，初始化节点
        self.head = ListNode(None)

    def is_empty(self):
        return self.head == None


    def addIndex(self, i, val): # 在特定位置插入一个元素
        node = ListNode(val)
        temp = self.head
        j = 0
        while temp.next != None :
            j += 1
            if j == i:
                node.next = temp.next
                temp.next = node
                break
            temp = temp.next


    def addAtHead(self, val): # 在头部插入一个元素
        node = ListNode(val)
        temp = self.head
        if self.is_empty():
            temp.next = node
        else:
            node.next = temp.next
            temp.next = node


    def addAtTail(self, val): # 尾部插入一个元素
        node = ListNode(val)
        temp = self.head
        while temp.next != None:
            temp = temp.next
        temp.next = node 


    def print_link(self):
        tmp = self.head
        while tmp.next != None:
            tmp = tmp.next
            print(tmp.val, end = ' ')
        print()


    def createLinkForward(self, nums):
        temp = self.head
        for i in nums:
            node = ListNode(i)
            node.next = temp.next
            temp.next = node


    def createLinkBackward(self, nums):
        temp = self.head
        for i in nums:
            node = ListNode(i)
            temp.next = node
            temp = node

    def reverse(self): # 反转列表
        temp = self.head
        res = []
        while temp.next != None:
            temp = temp.next
            res.append(temp.val)
        self.head = ListNode(None)

        self.createLinkForward(res)
        self.print_link()

a.addAtHead(1)
a.addAtHead(0)
```

```python
        res = []
        stack = []
        node = root

        while stack or node:
            while node:
                stack.append(node)
                node = node.left
            node = stack.pop()
            res.append(node.item)
            node = node.right
        return res

        res = []

            while node:

            node = stack.pop()

class treeDepth:
    def __init__(self):
        return

    def maxDepth(self, root):
        if not root:
            return 0
        return 1 + max(self.maxDepth(root.left), self.maxDepth(root.right))


    def bfs_Depth(self, root):
        if not root: return 0
        queue, res = [root], 0
        while queue:
            tmp = []
            for node in queue:
                if node.left: tmp.append(node.left)
                if node.right: tmp.append(node.right)

            queue = tmp
            res += 1

        return res
```
